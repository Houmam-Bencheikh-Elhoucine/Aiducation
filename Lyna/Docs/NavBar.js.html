import * as React from 'react';
import AppBar from '@mui/material/AppBar';
import Box from '@mui/material/Box';
import Toolbar from '@mui/material/Toolbar';
import Typography from '@mui/material/Typography';
import Button from '@mui/material/Button';
import { Stack } from '@mui/material';
/**
 * 
 * @function NavBar
 * @description This is a functional component in ReactJS that returns a NavBar component
 *  The NavBar is implemented using Material UI's AppBar component with a Toolbar component nested inside
 *  The NavBar has the following features:

 * A logo that takes up the majority of the space in the NavBar.
 * A row of buttons for "About", "Team", and "Contact Us".
 *  A button for "Sign in".
 *  The component uses the Typography component from Material UI to display the logo and Button components for the buttons. The "About", "Team", and "Contact Us" buttons are grouped using the Stack component with a spacing of 9. The "Sign in" button is placed on the right side of the NavBar.
 */

export default function NavBar() {
  return (
    
      <AppBar >
        <Toolbar>
          
          <Typography variant="h6" component="div" sx={{ flexGrow: 1 }}>
            LOGO
          </Typography>
          
          <Stack direction="row" spacing={9} sx={{flexGrow: 1}}>
          <Button variant='text' color="inherit">About</Button>
          <Button variant='text' color="inherit">Team</Button>
          <Button variant='text' color="inherit">Contact Us</Button>
          </Stack>
          
          <Button variant='text' color="inherit">Sign in</Button>
          
        </Toolbar>
      </AppBar>
   
  );
}
